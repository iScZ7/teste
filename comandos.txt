# Inicializar um repositório local (Pasta atual)
git init . #maluco no pedaço

# Verifica estado atual do repositório
git status

# Adicionar arquivos no stage
git add <nome>
git add .(tudo)

# Salvar estado do repositório
git commit -m "msg"

# Adicionar e commitar somente arquivos monitorados
git commit -am "msg"

# Restaurar mudanças no arquivo
git restore <nome>

# Remove arquivo do stage
git restore --staged <nome>

# Mostrar histórico do repositório
git log #maluco no pedaço
git log --oneline

# Mostrar alterações atuais
git diff

# Mostrar mudanças de um commit
git show <hash>

# Listar as branch
git branch

# Renomear uma branch
git branch -m <novo_nome>

# Criar nova branch
git branch <novo_nome>

# Trocar de branch
git switch <nome>

# Juntar uma branch na outra
git merge <nome>

# Apagar uma branch
git branch -d <nome>
git branch -D <nome>(força)

# Motrar quem alterou
git blame <nome_arquivo>

# Criar uma branch e ir para ela
git checkout -b <nome>

# Mover o HEAD para um commit
git checkout <hash>
