#alteração feita na web
#segunda mudança
# Inicializar um repositório local (Pasta atual)
git init . #phil

# Verifica estado atual do repositório
git status

# Clonar repositório
git clone

# Adicionar endereço remoto em um repositório local
git remote add <url>

# Verificar endereço remoto
git remote -v

# Enviar alterações e definir a upstream(remoto) com a branch local
git push -u origin <nome_branch>

# Enviar alterações para o github
git push 
git push origin <nome_branch>

# Enviar alterações de uma branch local para uma branch remota
git push origin <nome_branch_local> : <nome_branch_remota>

# Adicionar arquivos no stage
git add <nome>
git add .(tudo)

# Salvar estado do repositório
git commit -m "msg"

# Adicionar e commitar somente arquivos monitorados
git commit -am "msg"

#Commitar e substituir o arquivo
git commit - m "msg" --amend

# Restaurar mudanças no arquivo
git restore <nome>

# Remove arquivo do stage
git restore --staged <nome>

# Mostrar histórico do repositório
git log #phil
git log --oneline

# Mostrar alterações atuais
git diff

# Mostrar mudanças de um commit
git show <hash>

# Listar as branch
git branch

# Renomear uma branch
git branch -m <novo_nome>

# Criar nova branch
git branch <novo_nome>

# Trocar de branch
git switch <nome>

# Juntar uma branch na outra
git merge <nome>

# Apagar uma branch
git branch -d <nome>
git branch -D <nome>(força)

# Motrar quem alterou
git blame <nome_arquivo>

# Criar uma branch e ir para ela
git checkout -b <nome>

# Mover o HEAD para um commit
git checkout <hash>

# Voltar para um commit específico (local)
git reset <hash>
git reset --soft <hash> (mantem o stage salvo)
git reset --hard <hash> (apaga tudo)

# Voltar para um commit específico sem apagar histórico(remoto)
git revert <hash>

#baixar as atualizações do repositório remoto, sem mesclar
git fetch

#baixar as atualizações do repositório remoto e mesclar
git pull
